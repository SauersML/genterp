def refresh_import_paths():
    # AoU kernels put /opt/conda ahead of ~/.local. Pip --user installs then
    # succeed but imports still resolve stale conda packages. Put user-site
    # first so the packages this cell installs are the packages this cell uses.
    user_site = site.getusersitepackages()
    if user_site:
        if user_site in sys.path:
            sys.path.remove(user_site)
        sys.path.insert(1 if str(REPO) in sys.path else 0, user_site)
    for path in site.getsitepackages() if hasattr(site, "getsitepackages") else []:
        if path and path not in sys.path:
            site.addsitedir(path)
    importlib.invalidate_caches()

def stepfmt(x, _):
    if x >= 1e6:
        return f"{x / 1e6:.1f}M"
    if x >= 1e3:
        return f"{x / 1e3:.0f}k"
    return f"{int(x)}"


def ema(y, a=0.12):
    y = np.asarray(y, dtype=float)
    if len(y) == 0:
        return y
    out = np.empty_like(y)
    out[0] = y[0]
    for i in range(1, len(y)):
        out[i] = a * y[i] + (1 - a) * out[i - 1]
    return out


def artifact_rows(clt_dir):
    rows = []
    if not clt_dir.is_dir():
        return rows
    for path in clt_dir.iterdir():
        match = ARTIFACT_RE.match(path.name)
        if not match or not path.is_dir():
            continue
        cfg_file = path / "clt_config.json"
        state_file = path / "clt_training_state.json"
        if not cfg_file.is_file() or not state_file.is_file():
            continue
        state = json.loads(state_file.read_text())
        cfg = json.loads(cfg_file.read_text())
        metrics = state.get("metrics") or {}
        row = {
            "kind": match.group(1),
            "step": int(state.get("step", match.group(2))),
            "mtime": path.stat().st_mtime,
            "path": path,
            "cfg": cfg,
            "training_config": state.get("training_config") or {},
            "runtime": state.get("runtime") or {},
        }
        row.update({k: float(v) for k, v in metrics.items() if isinstance(v, (int, float))})
        rows.append(row)
    rows.sort(key=lambda r: (r["step"], r["mtime"]))
    return rows


def metric_rows(clt_dir):
    path = clt_dir / "clt_metrics.jsonl"
    if not path.is_file():
        return []
    rows = []
    for line_no, line in enumerate(path.read_text().splitlines(), start=1):
        if not line.strip():
            continue
        raw = json.loads(line)
        row = {"line_no": line_no}
        for key, value in raw.items():
            if key in {"kind", "step"}:
                row[key] = value
            elif isinstance(value, (int, float)):
                row[key] = float(value)
        if "step" in row:
            row["step"] = int(row["step"])
            rows.append(row)
    rows.sort(key=lambda r: (r["step"], r["line_no"]))
    return rows


def series(rows, key):
    xs, ys = [], []
    for row in rows:
        if key in row:
            xs.append(float(row["step"]))
            ys.append(float(row[key]))
    return np.asarray(xs), np.asarray(ys)


def standardize(x):
    return StandardScaler().fit_transform(np.asarray(x, dtype=np.float32))


def pca2(x):
    if len(x) < 2:
        return np.zeros((len(x), 2), dtype=np.float32), np.array([0.0, 0.0])
    pca = PCA(n_components=2, random_state=RNG_SEED)
    coords = pca.fit_transform(standardize(x))
    return coords, pca.explained_variance_ratio_


def umap2(x, *, neighbors=20, min_dist=0.08):
    if len(x) < 3:
        return np.zeros((len(x), 2), dtype=np.float32)
    n_neighbors = min(neighbors, max(2, len(x) - 1))
    return UMAP(
        n_components=2,
        n_neighbors=n_neighbors,
        min_dist=min_dist,
        metric="euclidean",
        random_state=RNG_SEED,
    ).fit_transform(standardize(x))


def top_indices(values, k):
    flat = np.asarray(values).reshape(-1)
    k = min(k, flat.size)
    if k <= 0:
        return np.array([], dtype=int)
    idx = np.argpartition(-flat, k - 1)[:k]
    return idx[np.argsort(-flat[idx])]


def strict_upper_pairs(n_layers):
    return [(s, t) for s in range(n_layers) for t in range(s + 1, n_layers)]


def decoder_block(state, pairs, diag_w, source_layer, target_layer):
    if source_layer == target_layer:
        return diag_w[source_layer]
    pair_index = pairs.index((source_layer, target_layer))
    if "off_blocks" in state:
        return state["off_blocks"][pair_index].detach().float().numpy()
    return (
        state["off_U"][pair_index].detach().float().numpy()
        @ state["off_V"][source_layer].detach().float().numpy()
    )


def knn_segments(coords, k=5):
    if len(coords) <= 2:
        return []
    nn = NearestNeighbors(n_neighbors=min(k + 1, len(coords))).fit(coords)
    _, idx = nn.kneighbors(coords)
    segments = []
    seen = set()
    for i, neigh in enumerate(idx):
        for j in neigh[1:]:
            a, b = sorted((int(i), int(j)))
            if (a, b) not in seen:
                seen.add((a, b))
                segments.append([coords[a], coords[b]])
    return segments


def move_batch(batch, device):
    return {k: v.to(device) if isinstance(v, torch.Tensor) else v for k, v in batch.items()}


def load_genterp_local(model_dir, device):
    model_dir = Path(model_dir)
    cfg = GenterpHFConfig.from_pretrained(model_dir)
    model = GenterpForCausalLM(cfg)
    state = {}
    safe_index = model_dir / "model.safetensors.index.json"
    if safe_index.is_file():
        weight_map = json.loads(safe_index.read_text())["weight_map"]
        for shard_name in sorted(set(weight_map.values())):
            state.update(load_safetensors(model_dir / shard_name, device="cpu"))
    elif (model_dir / "model.safetensors").is_file():
        state = load_safetensors(model_dir / "model.safetensors", device="cpu")
    elif (model_dir / "pytorch_model.bin").is_file():
        state = torch.load(model_dir / "pytorch_model.bin", map_location="cpu", weights_only=True)
    else:
        raise FileNotFoundError(f"no model weights found in {model_dir}")
    model.load_state_dict(state, strict=False)
    return model.to(device).eval()

node_features = np.column_stack([

node_score = dec_norm * np.log1p(enc_norm) * np.log1p(node_mean_activation) / np.sqrt(np.maximum(theta, EPS))
sample_node_flat = top_indices(node_score, min(NODE_UMAP_SAMPLE, node_score.size))

edges = []
for source_layer, target_layer in pairs:
    block = decoder_block(state, pairs, diag_w, source_layer, target_layer)
    source_candidates = top_indices(np.linalg.norm(block, axis=-1) * np.log1p(node_mean_activation[source_layer]), TOP_NODE_CANDIDATES)
    target_candidates = top_indices(enc_norm[target_layer] * np.log1p(node_mean_activation[target_layer]), TOP_NODE_CANDIDATES)
    if len(source_candidates) == 0 or len(target_candidates) == 0:
        continue
    scores = block[source_candidates] @ enc[target_layer, target_candidates].T
    take = min(GRAPH_EDGES, scores.size)
    flat = np.argpartition(-np.abs(scores).reshape(-1), take - 1)[:take]
    for item in flat:
        i = int(item // scores.shape[1])
        j = int(item % scores.shape[1])
        s_feat = int(source_candidates[i])
        t_feat = int(target_candidates[j])
        s_flat = source_layer * n_features + s_feat
        t_flat = target_layer * n_features + t_feat
        if np.isfinite(node_umap[s_flat]).all() and np.isfinite(node_umap[t_flat]).all():
            edges.append((s_flat, t_flat, float(scores[i, j])))
edges = sorted(edges, key=lambda e: abs(e[2]), reverse=True)[:GRAPH_EDGES]
