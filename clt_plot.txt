"""Paste into one notebook cell. Simple CLT-only artifact plots.

This cell deliberately does not import genterp, transformers, datasets, or the
Genterp model. It reads CLT artifacts directly:
  runs[-tiny]/clt/*/clt_config.json
  runs[-tiny]/clt/*/clt_training_state.json
  runs[-tiny]/clt/*/clt_state.pt

It plots:
  1. artifact / metrics trend
  2. node PCA and UMAP from learned CLT weights
  3. decoder-mediated node graph
  4. layer interaction heatmaps
  5. top-node table and bar chart
"""

import json
import re
import subprocess
import sys
from pathlib import Path

import matplotlib.pyplot as plt
import numpy as np


def require(import_name, pip_name=None):
    try:
        return __import__(import_name)
    except ModuleNotFoundError:
        if pip_name is None:
            raise
        print(f"installing missing plotting dependency: {pip_name}")
        subprocess.check_call([sys.executable, "-m", "pip", "install", pip_name])
        return __import__(import_name)


torch = require("torch", "torch")
try:
    from sklearn.decomposition import PCA
    from sklearn.preprocessing import StandardScaler
except ModuleNotFoundError:
    require("sklearn", "scikit-learn")
    from sklearn.decomposition import PCA
    from sklearn.preprocessing import StandardScaler

try:
    from umap import UMAP
except ModuleNotFoundError:
    require("umap", "umap-learn")
    from umap import UMAP

from matplotlib.colors import TwoSlopeNorm
from matplotlib.ticker import FuncFormatter, MaxNLocator


ARTIFACT_RE = re.compile(r"^(checkpoint|final)-(\d+)-[0-9a-f]+$")
ROOT = Path.home() / "genterp"
CLT_DIRS = [ROOT / "runs-tiny" / "clt", ROOT / "runs" / "clt"]
EDGE_QUANTILE = 0.999
EDGE_MIN_ALPHA = 0.10
EDGE_MAX_ALPHA = 0.82
EDGE_ALPHA_GAMMA = 1.7
RNG_SEED = 7
EPS = 1e-12
LAYER_COLORS = np.asarray(
    [
        "#2f6fbb",
        "#c2413b",
        "#2f8f5b",
        "#8a5fbf",
        "#c47a1f",
        "#1f8a9b",
        "#b23a7f",
        "#6f6f2f",
    ]
)


def step_fmt(x, _pos=None):
    x = float(x)
    if abs(x) >= 1_000_000:
        return f"{x / 1_000_000:.1f}M"
    if abs(x) >= 1_000:
        return f"{x / 1_000:.0f}k"
    return f"{x:.0f}"


def read_json(path):
    return json.loads(Path(path).read_text())


def artifact_rows(clt_dir):
    rows = []
    if not clt_dir.is_dir():
        return rows
    for path in clt_dir.iterdir():
        match = ARTIFACT_RE.match(path.name)
        if not match or not path.is_dir():
            continue
        cfg_path = path / "clt_config.json"
        train_path = path / "clt_training_state.json"
        state_path = path / "clt_state.pt"
        if not (cfg_path.is_file() and train_path.is_file() and state_path.is_file()):
            continue
        cfg = read_json(cfg_path)
        train_state = read_json(train_path)
        row = {
            "kind": match.group(1),
            "step": int(train_state.get("step", match.group(2))),
            "mtime": path.stat().st_mtime,
            "path": path,
            "cfg": cfg,
            "training_config": train_state.get("training_config") or {},
            "runtime": train_state.get("runtime") or {},
        }
        for key, value in (train_state.get("metrics") or {}).items():
            if isinstance(value, (int, float)):
                row[key] = float(value)
        rows.append(row)
    rows.sort(key=lambda r: (r["step"], r["mtime"], r["path"].name))
    return rows


def metric_rows(clt_dir):
    path = clt_dir / "clt_metrics.jsonl"
    rows = []
    if not path.is_file():
        return rows
    for line_no, line in enumerate(path.read_text().splitlines(), start=1):
        if not line.strip():
            continue
        raw = json.loads(line)
        row = {"line_no": line_no}
        for key, value in raw.items():
            if key in {"kind", "step"}:
                row[key] = value
            elif isinstance(value, (int, float)):
                row[key] = float(value)
        if "step" in row:
            row["step"] = int(row["step"])
            rows.append(row)
    rows.sort(key=lambda r: (r["step"], r["line_no"]))
    return rows


def numeric_series(rows, key):
    xs, ys = [], []
    for row in rows:
        if key in row:
            xs.append(float(row["step"]))
            ys.append(float(row[key]))
    return np.asarray(xs), np.asarray(ys)


def ema(y, alpha=0.18):
    y = np.asarray(y, dtype=float)
    if len(y) < 2:
        return y
    out = np.empty_like(y)
    out[0] = y[0]
    for i in range(1, len(y)):
        out[i] = alpha * y[i] + (1 - alpha) * out[i - 1]
    return out


def standardize(x):
    return StandardScaler().fit_transform(np.asarray(x, dtype=np.float32))


def pca2(x):
    x = np.asarray(x, dtype=np.float32)
    if len(x) < 2:
        return np.zeros((len(x), 2), dtype=np.float32), np.zeros(2)
    model = PCA(n_components=2, random_state=RNG_SEED)
    coords = model.fit_transform(standardize(x))
    return coords, model.explained_variance_ratio_


def umap2(x):
    x = np.asarray(x, dtype=np.float32)
    if len(x) < 4:
        return np.zeros((len(x), 2), dtype=np.float32)
    neighbors = min(30, max(2, len(x) - 1))
    return UMAP(
        n_components=2,
        n_neighbors=neighbors,
        min_dist=0.05,
        metric="euclidean",
        random_state=RNG_SEED,
    ).fit_transform(standardize(x))


def top_indices(values, k):
    values = np.asarray(values).reshape(-1)
    k = min(int(k), values.size)
    if k <= 0:
        return np.array([], dtype=int)
    idx = np.argpartition(-values, k - 1)[:k]
    return idx[np.argsort(-values[idx])]


def strict_upper_pairs(n_layers):
    return [(s, t) for s in range(n_layers) for t in range(s + 1, n_layers)]


def decoder_block(state, pairs, diag_w, source_layer, target_layer):
    if source_layer == target_layer:
        return diag_w[source_layer]
    pair_index = pairs.index((source_layer, target_layer))
    if "off_blocks" in state:
        return state["off_blocks"][pair_index].float().cpu().numpy()
    return (
        state["off_U"][pair_index].float().cpu().numpy()
        @ state["off_V"][source_layer].float().cpu().numpy()
    )


def node_label(flat, n_features):
    return f"L{int(flat) // n_features}F{int(flat) % n_features}"


def layer_legend(ax, layers_present):
    handles = []
    for layer in sorted(set(int(x) for x in layers_present)):
        handles.append(
            plt.Line2D(
                [0],
                [0],
                marker="o",
                linestyle="",
                markersize=7,
                markerfacecolor=LAYER_COLORS[layer % len(LAYER_COLORS)],
                markeredgecolor="white",
                markeredgewidth=0.7,
                label=f"layer {layer}",
            )
        )
    ax.legend(handles=handles, title="CLT layer", loc="best")


def choose_run():
    candidates = [(path, artifact_rows(path)) for path in CLT_DIRS]
    candidates = [(path, rows) for path, rows in candidates if rows]
    if not candidates:
        raise SystemExit(f"no CLT artifacts found under {[str(p) for p in CLT_DIRS]}")
    return max(candidates, key=lambda item: item[1][-1]["mtime"])


clt_dir, artifacts = choose_run()
metrics = metric_rows(clt_dir)
history = metrics if len(metrics) >= 2 else artifacts
latest = artifacts[-1]
cfg = latest["cfg"]
tcfg = latest["training_config"]
runtime = latest["runtime"]

state = torch.load(latest["path"] / "clt_state.pt", map_location="cpu", weights_only=True)
enc = state["enc_weight"].float().cpu().numpy()
enc_bias = state.get("enc_bias")
diag_w = state["diag_W"].float().cpu().numpy()
dec_bias = state.get("dec_bias")
theta = np.exp(state["log_threshold"].float().cpu().numpy())

layers, n_features, dim = enc.shape
pairs = strict_upper_pairs(layers)
layer_id = np.repeat(np.arange(layers), n_features)

enc_norm = np.linalg.norm(enc, axis=-1)
dec_norm = np.linalg.norm(diag_w, axis=-1)
alignment = np.sum(enc * diag_w, axis=-1) / np.maximum(enc_norm * dec_norm, EPS)
theta_inv = 1.0 / np.sqrt(np.maximum(theta, EPS))

if enc_bias is not None:
    enc_bias_np = enc_bias.float().cpu().numpy()
else:
    enc_bias_np = np.zeros((layers, n_features), dtype=np.float32)

node_features = np.column_stack(
    [
        np.log1p(enc_norm.reshape(-1)),
        np.log1p(dec_norm.reshape(-1)),
        np.log(np.maximum(theta.reshape(-1), EPS)),
        alignment.reshape(-1),
        enc_bias_np.reshape(-1),
        layer_id / max(layers - 1, 1),
    ]
)
node_pca, pca_var = pca2(node_features)
node_umap = umap2(node_features)

plt.rcParams.update(
    {
        "figure.facecolor": "white",
        "axes.facecolor": "#fafafa",
        "axes.edgecolor": "#cccccc",
        "axes.spines.top": False,
        "axes.spines.right": False,
        "axes.grid": True,
        "grid.color": "#e6e6e6",
        "grid.linewidth": 0.7,
        "axes.titleweight": "semibold",
        "axes.titlepad": 9,
        "legend.frameon": False,
        "font.family": "DejaVu Sans",
    }
)

print("=" * 96)
print(f"clt dir:   {clt_dir}")
print(f"artifact:  {latest['path'].name}    step={latest['step']:,}    artifacts={len(artifacts):,}    metric_rows={len(metrics):,}")
print(
    "config:    "
    f"layers={layers} dim={dim} features/layer={n_features} "
    f"rank={cfg.get('off_diagonal_rank')} sparsity_coef={cfg.get('sparsity_coef')}"
)
if tcfg:
    print(

        f"new_steps/run={tcfg.get('steps')} lr={tcfg.get('learning_rate')} "
        f"tokens/chunk={tcfg.get('activation_batch_tokens')} eval_every={tcfg.get('eval_every')}"
    )
if runtime:
    print(f"runtime:   {runtime.get('device_type')}:{runtime.get('device_index')} precision={runtime.get('precision')}")
print("-" * 96)
for key in ["loss", "recon", "sparsity", "n_active", "grad_norm", "eval_loss", "eval_recon", "eval_sparsity", "eval_n_active"]:
    if key in latest:
        print(f"{key:14s} {latest[key]:>12.6g}")
print("=" * 96)

print("\nCLT nodes with largest decoder norm")
print(f"{'rank':>4s} {'node':>8s} {'dec_norm':>10s} {'enc_norm':>10s} {'theta':>10s} {'cos':>8s}")
top_decoder_nodes = top_indices(dec_norm.reshape(-1), min(24, dec_norm.size))
for rank, flat in enumerate(top_decoder_nodes, start=1):
    layer = int(flat // n_features)
    feature = int(flat % n_features)
    print(
        f"{rank:4d} {node_label(flat, n_features):>8s} {dec_norm[layer, feature]:10.4g} "
        f"{enc_norm[layer, feature]:10.4g} {theta[layer, feature]:10.4g} "
        f"{alignment[layer, feature]:8.3f}"
    )


# 1. Training / artifact trend. This always plots the best available history.
fig, axes = plt.subplots(2, 2, figsize=(15, 9), constrained_layout=True)
fig.suptitle("CLT Training", fontsize=14, fontweight="bold")

# 3. Decoder-mediated graph on UMAP.
edges = []
for source_layer, target_layer in pairs:
    block = decoder_block(state, pairs, diag_w, source_layer, target_layer)
    scores = block @ enc[target_layer].T
    strength = np.abs(scores)
    max_strength = float(strength.max()) if strength.size else 0.0
    if max_strength <= EPS:
        continue
    cutoff = float(np.quantile(strength.reshape(-1), EDGE_QUANTILE))
    source_features, target_features = np.where(strength >= cutoff)
    for s_feat, t_feat in zip(source_features, target_features, strict=True):
        s_flat = source_layer * n_features + s_feat
        t_flat = target_layer * n_features + t_feat
        importance = (strength[s_feat, t_feat] - cutoff) / max(max_strength - cutoff, EPS)
        importance = float(np.clip(importance, 0.0, 1.0) ** EDGE_ALPHA_GAMMA)
        edges.append((s_flat, t_flat, float(scores[s_feat, t_feat]), importance))

fig, ax = plt.subplots(figsize=(12, 9), constrained_layout=True)
graph_sc = ax.scatter(
    node_umap[:, 0],
    node_umap[:, 1],
    c=decoder_color,
    s=node_size,
    cmap="viridis",
    alpha=0.46,
    edgecolors="white",
    linewidths=0.25,
)
if edges:
    starts = np.asarray([node_umap[s_flat] for s_flat, _t_flat, _weight, _alpha in edges])
    stops = np.asarray([node_umap[t_flat] for _s_flat, t_flat, _weight, _alpha in edges])
    weights = np.asarray([weight for _s_flat, _t_flat, weight, _alpha in edges])
    alphas = np.asarray([alpha for _s_flat, _t_flat, _weight, alpha in edges])
    alphas = EDGE_MIN_ALPHA + (EDGE_MAX_ALPHA - EDGE_MIN_ALPHA) * alphas
    widths = 0.0013 + 0.0042 * np.asarray([alpha for _s_flat, _t_flat, _weight, alpha in edges])
    colors = np.zeros((len(edges), 4), dtype=float)
    colors[weights >= 0] = np.array([198 / 255, 40 / 255, 40 / 255, 1.0])
    colors[weights < 0] = np.array([21 / 255, 101 / 255, 192 / 255, 1.0])
    colors[:, 3] = np.clip(alphas, EDGE_MIN_ALPHA, EDGE_MAX_ALPHA)
    deltas = stops - starts
    ax.quiver(
        starts[:, 0],
        starts[:, 1],
        deltas[:, 0],
        deltas[:, 1],
        angles="xy",
        scale_units="xy",
        scale=1,
        color=colors,
        width=widths,
        headwidth=5.2,
        headlength=7.2,
        headaxislength=5.8,
        minlength=0.0,
        pivot="tail",
        zorder=1,
    )
fig.colorbar(graph_sc, ax=ax, label="log10 decoder norm")
ax.set_title("Directed Edges", loc="left")
ax.set_xlabel("node UMAP1")
ax.set_ylabel("node UMAP2")
plt.show()


# 4. Layer interactions and node diagnostics.
layer_interaction = np.zeros((layers, layers), dtype=np.float64)
for source_layer in range(layers):
    for target_layer in range(layers):
        if target_layer < source_layer:
            continue
        block = decoder_block(state, pairs, diag_w, source_layer, target_layer)
        layer_interaction[source_layer, target_layer] = np.linalg.norm(block)

fig, axes = plt.subplots(2, 2, figsize=(15, 10), constrained_layout=True)
fig.suptitle("CLT Weights", fontsize=14, fontweight="bold")

for source_layer in range(layers):

        ax.text(target_layer, source_layer, f"{layer_interaction[source_layer, target_layer]:.2g}", ha="center", va="center", fontsize=9)
