"""Paste into a notebook cell. Runs `genterp.eval_cindex` if results don't
already exist on disk, then plots a forest plot of Harrell's C-index per
disease with bootstrap 95% confidence intervals.

Auto-detects tiny vs full: if `runs-tiny/cindex_arrays.npz` exists it uses
that; if `runs/cindex_arrays.npz` exists it uses that. If neither, runs the
eval on whichever model has a `final_checkpoint.json` (preferring tiny).

Reads bootstrap CIs / cohort metadata straight from cindex_results.json
(eval already does 500 resamples per disease) — no in-notebook re-bootstrap.
The arrays npz is used only if you want to recompute Harrell's C from raw.
"""

import json
import subprocess
import time
from pathlib import Path

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.ticker import MaxNLocator

CANDIDATE_DIRS = [Path.home() / "genterp" / "runs-tiny", Path.home() / "genterp" / "runs"]


REQUIRED_NPZ_KEYS = ("disease_names", "risks", "time_to_event", "observed",
                     "prior_case", "sex_eligible", "landmark_age_days")


def _npz_has_all_keys(path: Path) -> bool:
    """Stale eval outputs (pre per-subject-landmark, pre sex-eligible) lack
    keys the plot now expects. Treat such files as "missing" so we re-run."""
    try:
        with np.load(path, allow_pickle=False) as f:
            return all(k in f.files for k in REQUIRED_NPZ_KEYS)
    except Exception:
        return False


def _run_eval_if_needed():
    """If no fresh cindex_arrays.npz exists, find a trained model and run the eval.
    A "fresh" file has every key in REQUIRED_NPZ_KEYS — stale files (from older
    eval versions) trigger a delete + re-run so the plot doesn't KeyError."""
    for d in CANDIDATE_DIRS:
        npz = d / "cindex_arrays.npz"
        if npz.is_file():
            if _npz_has_all_keys(npz):
                return
            print(f"  stale {npz} (missing new fields); deleting and re-running eval")
            npz.unlink()
            (d / "cindex_results.json").unlink(missing_ok=True)
    runs_tiny = Path.home() / "genterp" / "runs-tiny"
    runs_full = Path.home() / "genterp" / "runs"
    if (runs_tiny / "final_checkpoint.json").is_file():
        flag = ["--tiny"]
        target = "tiny"
    elif (runs_full / "final_checkpoint.json").is_file():
        flag = []
        target = "full"
    else:
        raise SystemExit(
            "no trained final-model under runs-tiny/ or runs/ — "
            "run `~/genterp/run.sh [--tiny]` to train first."
        )
    # Notebook kernel = system Python (no torch). Project lives in ~/genterp
    # with a uv-managed .venv that has torch. Pull origin/main first so the
    # eval picks up the latest fixes (notebook bypasses run.sh's self-update).
    repo = str(Path.home() / "genterp")
    print("  refreshing repo to origin/main before eval...")
    subprocess.run(["git", "fetch", "--quiet", "origin", "main"], cwd=repo, check=True)
    subprocess.run(["git", "reset", "--quiet", "--hard", "origin/main"], cwd=repo, check=True)
    subprocess.run(["uv", "sync", "--quiet"], cwd=repo, check=True)
    cmd = ["uv", "run", "python", "-m", "genterp.eval_cindex", *flag]
    print(f"  no eval results on disk — running `{' '.join(cmd)}` ({target})")
    print("  this can take several minutes; output streams below ↓\n")
    rc = subprocess.run(cmd, cwd=repo).returncode
    if rc != 0:
        raise SystemExit(f"eval_cindex exited with status {rc}")


def latest_eval_dir(dirs):
    candidates = []
    for d in dirs:
        f = d / "cindex_arrays.npz"
        if f.is_file():
            candidates.append((f.stat().st_mtime, d))
    if not candidates:
        raise SystemExit(
            "still no cindex_arrays.npz after eval attempt — check error output above."
        )
    return max(candidates)[1]


_run_eval_if_needed()

run_dir = latest_eval_dir(CANDIDATE_DIRS)
mtime_min = (time.time() - (run_dir / "cindex_arrays.npz").stat().st_mtime) / 60
is_tiny = run_dir.name == "runs-tiny"

meta = json.loads((run_dir / "cindex_results.json").read_text())
arrays = np.load(run_dir / "cindex_arrays.npz", allow_pickle=False)
disease_names = [str(s) for s in arrays["disease_names"].tolist()]
risks = arrays["risks"]                      # (n_subjects, n_diseases)
time_to_event = arrays["time_to_event"]       # (n_subjects, n_diseases)
observed = arrays["observed"]                  # (n_subjects, n_diseases)
prior_case = arrays["prior_case"]             # (n_subjects, n_diseases)
sex_eligible = arrays["sex_eligible"]          # (n_subjects, n_diseases)
landmark_age_days = arrays["landmark_age_days"]  # (n_subjects,)
landmark_age_y = landmark_age_days / 365.25

# ───────── header dump ─────────
bar = "═" * 92
sep = "─" * 92
phen = meta.get("phenotype_info", {})
lm_summary = meta.get("landmark_summary", {})
print(bar)
print(f"  source:        {run_dir}/cindex_arrays.npz   ({mtime_min:.1f} min ago)")
print(f"  run mode:      {'TINY' if is_tiny else 'FULL'}")
print(f"  final model:   {meta['final_model']}")
print(f"  landmark age:  per-subject in "
      f"[{meta.get('landmark_age_min_years', 40):.0f}, {meta.get('landmark_age_max_years', 85):.0f}] y  "
      f"(cohort: mean {lm_summary.get('mean_years', 0):.1f}y, "
      f"p10 {lm_summary.get('p10_years', 0):.1f}y → p90 {lm_summary.get('p90_years', 0):.1f}y)")
print(f"  horizon:       {meta['horizon_years']:.1f} y")
print(f"  max gap:       {meta['max_gap_years']:.1f} y")
print(f"  min hist:      {meta.get('min_pre_landmark_history_days', 180):.0f} d pre-landmark")
print(f"  n_eligible:    {meta['n_subjects_eligible']:,}")
print(f"  n_diseases:    {len(disease_names)}")
print(bar)

# ───────── per-disease rows from results JSON ─────────
rows = []
for d_idx, name in enumerate(disease_names):
    m = meta["results"].get(name, {})
    info = phen.get(name, {})
    rows.append({
        "name": name,
        "sex_restriction": info.get("sex_restriction"),
        "n_atoms": info.get("n_atoms"),
        "prior_cases": m.get("prior_cases", 0),
        "sex_excluded": m.get("sex_excluded", 0),
        "n_eligible": m.get("n_eligible", 0),
        "n_events": m.get("events", 0),
        "incidence_pct": m.get("incidence_pct", 0.0),
        "c_point": m.get("c_index"),
        "c_lo": m.get("c_index_lo"),
        "c_hi": m.get("c_index_hi"),
        "n_pairs": m.get("n_pairs", 0),
    })

name_w = max(len(r["name"]) for r in rows)
print(f"  {'disease':<{name_w}}  {'sex':>3}  {'n_elig':>7}  {'evt':>4}  {'inc%':>5}  {'C (95% CI)':<26}")
print(sep)
for r in sorted(rows, key=lambda x: -(x["c_point"] or 0)):
    sex_tag = r["sex_restriction"] or "-"
    if r["c_point"] is None:
        c_str = "  nan"
    elif r["c_lo"] is not None and r["c_hi"] is not None:
        c_str = f"{r['c_point']:.3f}  [{r['c_lo']:.3f}, {r['c_hi']:.3f}]"
    else:
        c_str = f"{r['c_point']:.3f}"
    print(f"  {r['name']:<{name_w}}  {sex_tag:>3}  {r['n_eligible']:>7,}  "
          f"{r['n_events']:>4,}  {r['incidence_pct']:>4.1f}  {c_str}")

summary = meta.get("summary")
if summary:
    print(sep)
    print(f"  cindex_mean (≥{summary['min_events_threshold']} events): "
          f"{summary['cindex_mean_well_powered']:.4f}  "
          f"over {summary['n_well_powered_diseases']} well-powered diseases")
print(bar + "\n")

# ───────── two-panel plot ─────────
plt.rcParams.update({
    "figure.facecolor": "white", "axes.facecolor": "#fafafa",
    "axes.edgecolor": "#cccccc", "axes.spines.top": False, "axes.spines.right": False,
    "axes.grid": True, "grid.color": "#e6e6e6", "grid.linewidth": 0.7,
    "axes.titleweight": "semibold", "axes.titlepad": 12,
    "legend.frameon": False, "font.family": "DejaVu Sans",
})

plot_rows = sorted(rows, key=lambda x: (x["c_point"] if x["c_point"] is not None else -np.inf))
n = len(plot_rows)
y = np.arange(n)
n_events = np.array([r["n_events"] for r in plot_rows])

fig = plt.figure(figsize=(15, max(5, 0.55 * n + 3.5)), constrained_layout=True)
gs = fig.add_gridspec(2, 2, height_ratios=[3, 1], width_ratios=[3, 1])
ax_forest = fig.add_subplot(gs[0, 0])
ax_inc = fig.add_subplot(gs[0, 1])
ax_lm = fig.add_subplot(gs[1, :])

# Colour by event count (well-powered → bright; underpowered → dim).
norm = plt.Normalize(vmin=max(n_events.min(), 1), vmax=max(n_events.max(), 2))
cmap = plt.get_cmap("viridis")
colors = [cmap(norm(ne)) for ne in n_events]

# Forest plot
ax_forest.axvline(0.5, color="#888", lw=0.9, ls="--", alpha=0.7, label="chance (0.5)")
for i, r in enumerate(plot_rows):
    if r["c_point"] is None:
        ax_forest.text(0.5, i, "  insufficient events", va="center", ha="left",
                       color="#aa6666", fontsize=9)
        continue
    if r["c_lo"] is not None and r["c_hi"] is not None:
        ax_forest.plot([r["c_lo"], r["c_hi"]], [i, i], lw=2.0, color=colors[i],
                       solid_capstyle="round")
    ax_forest.plot(r["c_point"], i, "o", color=colors[i], ms=10,
                   mec="white", mew=1.5, zorder=3)
labels = []
for r in plot_rows:
    sex_tag = f" {r['sex_restriction']}" if r["sex_restriction"] else ""
    labels.append(f"{r['name']}{sex_tag}  (n={r['n_eligible']:,}, k={r['n_events']:,})")
ax_forest.set_yticks(y); ax_forest.set_yticklabels(labels, fontsize=10)
ax_forest.set_xlabel("Harrell's C-index  (right-censored, 10-year horizon)")
ax_forest.set_xlim(0.30, 1.0)
ax_forest.set_title(
    f"Forest plot — {meta['results'][next(iter(meta['results']))].get('bootstrap_resamples', 0)} "
    f"bootstrap resamples per disease",
    loc="left",
)
ax_forest.legend(loc="lower right")

# Incidence panel
inc = np.array([r["incidence_pct"] for r in plot_rows])
ax_inc.barh(y, inc, color=colors, edgecolor="white", linewidth=0.6)
for i, v in enumerate(inc):
    ax_inc.text(v + 0.2, i, f"{v:.1f}%", va="center", fontsize=9, color="#444")
ax_inc.set_yticks(y); ax_inc.set_yticklabels([])
ax_inc.set_xlabel("10-year incidence in eligible cohort (%)")
ax_inc.set_title("Outcome rate", loc="left")
ax_inc.xaxis.set_major_locator(MaxNLocator(5))

# Landmark age distribution — across all eligible subjects
ax_lm.hist(landmark_age_y, bins=40, color="#5b8def", alpha=0.85, edgecolor="white")
ax_lm.axvline(landmark_age_y.mean(), color="#d62728", lw=1.5, ls="--",
              label=f"mean = {landmark_age_y.mean():.1f} y")
ax_lm.axvline(np.median(landmark_age_y), color="#2ca02c", lw=1.5, ls=":",
              label=f"median = {np.median(landmark_age_y):.1f} y")
ax_lm.set_xlabel("Per-subject landmark age (years)")
ax_lm.set_ylabel("subjects")
ax_lm.set_title("Landmark age distribution across the eligible cohort", loc="left")
ax_lm.legend(loc="best")

mode_tag = "TINY" if is_tiny else "FULL"
fig.suptitle(
    f"Genterp 10-year incident-disease C-index — {mode_tag}\n"
    f"per-subject landmark ∈ [{meta.get('landmark_age_min_years', 40):.0f}, "
    f"{meta.get('landmark_age_max_years', 85):.0f}] y, "
    f"horizon={meta['horizon_years']:.0f}y, N={meta['n_subjects_eligible']:,}",
    fontsize=13, fontweight="bold",
)
plt.show()
